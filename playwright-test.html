/**
 * Page Object Model for the OrangeHRM Login Page.
 *
 * Encapsulates all locators and actions related to the login screen.
 */
class LoginPage {
  /**
   * @param {import('@playwright/test').Page} page
   */
  constructor(page) {
    this.page = page;

    // --- Locators ---
    this.usernameInput = page.locator('input[name="username"]');
    this.passwordInput = page.locator('input[name="password"]');
    this.loginButton   = page.locator('button[type="submit"]');
    this.errorMessage  = page.locator(".oxd-alert-content--error");
    this.brandingLogo  = page.locator(".orangehrm-login-branding img");
  }

  // --- Actions ---

  /** Navigate to the login page. */
  async navigate() {
    await this.page.goto("/web/index.php/auth/login");
    await this.brandingLogo.waitFor({ state: "visible" });
  }

  /**
   * Perform login with provided credentials.
   * @param {string} username
   * @param {string} password
   */
  async login(username, password) {
    await this.usernameInput.fill(username);
    await this.passwordInput.fill(password);
    await this.loginButton.click();
  }

  /** Returns the error message text if login fails. */
  async getErrorMessage() {
    await this.errorMessage.waitFor({ state: "visible" });
    return this.errorMessage.textContent();
  }
}

module.exports = { LoginPage };

/**
 * Page Object Model for the OrangeHRM Dashboard Page.
 *
 * Handles top-level navigation (sidebar menu items) and logout.
 */
class DashboardPage {
  /**
   * @param {import('@playwright/test').Page} page
   */
  constructor(page) {
    this.page = page;

    // --- Locators ---
    this.sidebarMyInfo    = page.locator('a.oxd-main-menu-item',
                              { hasText: "My Info" });
    this.userDropdown     = page.locator(".oxd-userdropdown-tab");
    this.logoutLink       = page.locator(
      'a[href="/web/index.php/auth/logout"]');
    this.dashboardHeading = page.locator("h6",
                              { hasText: "Dashboard" });
  }

  /** Wait until the dashboard is fully loaded. */
  async waitForDashboard() {
    await this.dashboardHeading.waitFor({
      state: "visible", timeout: 15000
    });
  }

  /** Click "My Info" in the left sidebar. */
  async navigateToMyInfo() {
    await this.sidebarMyInfo.click();
    await this.page.waitForURL("**/viewPersonalDetails/**",
      { timeout: 15000 });
  }

  /** Perform logout via the user dropdown. */
  async logout() {
    await this.userDropdown.click();
    await this.logoutLink.waitFor({ state: "visible" });
    await this.logoutLink.click();
    await this.page.waitForURL("**/auth/login",
      { timeout: 15000 });
  }
}

module.exports = { DashboardPage };

/**
 * Page Object Model for the OrangeHRM Dependents tab
 * under My Info → Dependents.
 */
class DependentsPage {
  constructor(page) {
    this.page = page;

    // --- Tab Locator ---
    this.dependentsTab = page.locator(
      'a[href*="viewDependents"]', { hasText: "Dependents" });

    // --- Form Locators ---
    // Get the Add button specifically from the Assigned Dependents section (first Add button)
    this.addButton    = page.locator("button", { hasText: "Add" }).first();
    this.saveButton   = page.locator(
      'button[type="submit"]', { hasText: "Save" });
    this.cancelButton = page.locator("button", { hasText: "Cancel" });

    // Form fields inside .oxd-form
    this.nameInput = page.locator(
      '.oxd-form .oxd-input-group:has(label:text("Name")) input');
    this.relationshipDropdown = page.locator(
      '.oxd-form .oxd-input-group:has(label:text("Relationship")) .oxd-select-text');
    this.dateOfBirthInput = page.locator(
      '.oxd-form .oxd-input-group:has(label:text("Date of Birth")) input');

    // --- Table Locators ---
    this.dependentsTable = page.locator(".oxd-table");
    this.tableRows       = page.locator(
      ".oxd-table .oxd-table-body .oxd-table-row");
    this.toastMessage    = page.locator(".oxd-toast");

    // --- Delete Locators ---
    this.deleteCheckbox      = page.locator(
      ".oxd-table-body .oxd-table-row .oxd-checkbox-input");
    this.deleteSelectedButton = page.locator(
      "button", { hasText: "Delete Selected" });
    this.confirmDeleteButton  = page.locator(
      ".orangehrm-modal-footer button",
      { hasText: "Yes, Delete" });
  }

  /** Navigate to the Dependents tab within My Info. */
  async navigateToDependentsTab() {
    await this.dependentsTab.click();
    await this.page.waitForURL("**/viewDependents/**",
      { timeout: 15000 });
    await this.addButton.waitFor({
      state: "visible", timeout: 10000
    });
  }

  /** Get the count of dependents currently in the table. */
  async getDependentCount() {
    await this.page.waitForTimeout(1000);
    return await this.tableRows.count();
  }

  /**
   * Add a new dependent.
   * @param {string} name
   * @param {string} relationship  - "Child", "Spouse", or "Other"
   * @param {string} [dateOfBirth] - YYYY-MM-DD (optional)
   */
  async addDependent(name, relationship, dateOfBirth) {
    await this.addButton.click();

    // Fill the name
    await this.nameInput.waitFor({ state: "visible", timeout: 5000 });
    await this.nameInput.fill(name);

    // Select relationship from dropdown
    await this.relationshipDropdown.click();
    const option = this.page.locator(
      `.oxd-select-dropdown .oxd-select-option`,
      { hasText: relationship }
    );
    await option.click();

    // Fill date of birth if provided
    if (dateOfBirth) {
      await this.dateOfBirthInput.clear();
      await this.dateOfBirthInput.fill(dateOfBirth);
      await this.page.keyboard.press("Escape");
    }

    // Save & wait for success toast
    await this.saveButton.click();
    await this.toastMessage.waitFor({
      state: "visible", timeout: 10000
    });
  }

  /** Check whether a dependent with the given name exists. */
  async isDependentPresent(name) {
    await this.page.waitForTimeout(1000);
    const row = this.page.locator(
      `.oxd-table-body .oxd-table-row .oxd-table-cell:has-text("${name}")`
    );
    return (await row.count()) > 0;
  }

  /** Delete the first dependent using checkbox + Delete Selected. */
  async deleteFirstDependent() {
    const firstCheckbox = this.deleteCheckbox.first();
    await firstCheckbox.click();
    await this.deleteSelectedButton.click();
    await this.confirmDeleteButton.waitFor({ state: "visible" });
    await this.confirmDeleteButton.click();
    await this.toastMessage.waitFor({
      state: "visible", timeout: 10000
    });
  }
}

module.exports = { DependentsPage };

const { test, expect } = require("@playwright/test");
const { LoginPage } = require("../pages/LoginPage");
const { DashboardPage } = require("../pages/DashboardPage");
const { DependentsPage } = require("../pages/DependentsPage");
const testData = require("../utils/testData");

test.describe("Orange HRM - Dependents Management", () => {
  test("should complete full dependents workflow: Login > My Info > Dependents > Add > Verify > Logout", async ({ page }) => {
    // Initialize page objects
    const loginPage = new LoginPage(page);
    const dashboardPage = new DashboardPage(page);
    const dependentsPage = new DependentsPage(page);

    // Step 1: Open login page
    console.log("Step 1: Opening login page...");
    await loginPage.navigate();

    // Step 2: Login
    console.log("Step 2: Logging in with valid credentials...");
    await loginPage.login(testData.validLogin.username, testData.validLogin.password);
    await dashboardPage.waitForDashboard();
    expect(page.url()).toContain("dashboard");

    // Step 3: Click My Info
    console.log("Step 3: Navigating to My Info...");
    await dashboardPage.navigateToMyInfo();
    await page.waitForTimeout(1000);

    // Step 4: Click Dependents
    console.log("Step 4: Navigating to Dependents tab...");
    await dependentsPage.navigateToDependentsTab();
    expect(page.url()).toContain("viewDependents");

    // Step 5: Add a dependent
    console.log("Step 5: Adding a new dependent...");
    const initialCount = await dependentsPage.getDependentCount();
    const dependent = testData.dependents[0];

    await dependentsPage.addDependent(
      dependent.name,
      dependent.relationship,
      dependent.dateOfBirth
    );
    await page.waitForTimeout(2000); // Wait for table to update

    // Step 6: Verify dependent is added
    console.log("Step 6: Verifying dependent was added...");
    const isPresent = await dependentsPage.isDependentPresent(dependent.name);
    expect(isPresent).toBeTruthy();

    const finalCount = await dependentsPage.getDependentCount();
    expect(finalCount).toBeGreaterThan(initialCount);

    // Step 7: Perform Logout
    console.log("Step 7: Performing logout...");
    await dashboardPage.logout();
    expect(page.url()).toContain("login");
    console.log("Workflow completed successfully!");
  });
});

Assignment Review — Questions & Answers

A — Page Object Model Concepts

  • Q01What is the Page Object Model and why is it used in test automation?▾
    The Page Object Model (POM) is a design pattern where each web page (or major component) is represented by its own class. That class holds all the locators (selectors for elements) and action methods (login, addDependent, etc.) for that page. Tests call page-object methods instead of touching raw selectors.

    Benefits: (1) Maintainability — if the UI changes, you update one file, not every test. (2) Readability — tests read like business steps: loginPage.login(user, pass). (3) Reusability — multiple tests share the same page objects. (4) Separation of concerns — test logic is separate from page structure.
  • Q02How did you structure your page objects? Walk us through the folder layout.▾
    Three main folders: pages/ contains one class per page — LoginPage.js, DashboardPage.js, and DependentsPage.js. tests/ holds the spec file that orchestrates the flow. utils/ stores shared test data (credentials, dependent info, URLs) in testData.js.

    Each page class receives the Playwright page object via its constructor. Locators are defined as properties in the constructor, and actions are async methods. The test file imports all three page classes plus test data, instantiates them in beforeEach, and chains their methods.
  • Q03Why separate locators and actions into page classes instead of the test file?▾
    If locators live inside tests, a single UI change (e.g. a button class rename) could break dozens of test files. With POM, you fix it in one place. Page classes also encapsulate how to interact with the page, while tests describe what the business scenario is. This makes tests shorter, more readable, and easier for non-developers to understand.
  • Q04If a UI element changes, how many files need updating with POM vs without?▾
    With POM: Exactly 1 file — the relevant page object. For example, if the login button selector changes, you update only LoginPage.js.

    Without POM: Every test file that references that selector — potentially 10, 20, or more files. This is the single biggest maintenance advantage of POM.
  • Q05How do your page objects communicate — does one call another, or does the test orchestrate?▾
    The test orchestrates. Page objects are independent — LoginPage doesn't know about DashboardPage. The spec file creates all page objects and calls their methods in sequence. This keeps page objects loosely coupled and reusable in any test that needs them.

B — Locator Strategy

  • Q06What locator strategies did you use (CSS, XPath, text, role, test-id)? Why?▾
    Primarily CSS selectors combined with Playwright's hasText filter. For example: page.locator('a.oxd-main-menu-item', { hasText: "My Info" }). CSS selectors are fast, well-supported, and readable. The hasText option adds semantic meaning so the selector is both structurally and textually specific. I also used attribute selectors like input[name="username"] where stable HTML attributes were available.
  • Q07OrangeHRM lacks data-testid attributes — how did you handle that?▾
    Without data-testid, I relied on: (1) Stable HTML attributes like name, type, and href. (2) CSS class names from the OXD component library (e.g. .oxd-userdropdown-tab). (3) Parent-child relationships with text filters — e.g. .oxd-input-group:has(label:text("Name")) input finds the input associated with a specific label. This "label-anchored" approach is resilient because label text rarely changes without the feature itself changing.
  • Q08How would you deal with a locator that matches multiple elements?▾
    Several options: (1) Make the selector more specific by adding parent context, e.g. .oxd-form .oxd-input-group:has(label:text("Name")) input. (2) Use .first(), .nth(index), or .last() to target a specific match. (3) Filter with { hasText: "..." } or { has: locator } to narrow results. (4) Use locator.filter() for chained conditions. The goal is always to have a unique match for reliability.
  • Q09Difference between page.locator() and page.getByRole() / getByText()?▾
    page.locator() accepts any CSS or XPath selector — it's the most flexible. getByRole(), getByText(), getByLabel() are user-facing locators that target elements the way a user perceives them (by ARIA role, visible text, or label).

    Best practice: Prefer getByRole/getByLabel when the app has good accessibility markup. Fall back to locator() with CSS when ARIA roles are missing — which is the case with OrangeHRM's custom OXD components.
  • Q10How did you locate the Relationship dropdown and select a value?▾
    OrangeHRM uses a custom dropdown (not a native <select>), so selectOption() won't work. The approach: (1) Click the dropdown trigger: .oxd-input-group:has(label:text("Relationship")) .oxd-select-text. (2) Wait for the panel to appear. (3) Click the matching option: .oxd-select-dropdown .oxd-select-option filtered with { hasText: "Child" }. This two-step "click-then-select" pattern is standard for custom/React dropdowns.

C — Waits & Synchronisation

  • Q11Does Playwright use implicit or explicit waits? How does auto-wait work?▾
    Playwright uses auto-wait, which is neither traditional implicit nor explicit. When you call an action like .click() or .fill(), Playwright automatically waits for the element to be: visible, enabled, stable (not animating), able to receive events, and attached to the DOM. This happens up to the configured actionTimeout. It eliminates most manual sleep calls required in Selenium.
  • Q12Where did you add explicit waits, and why were they necessary?▾
    I used waitForURL() after navigation actions (clicking "My Info", "Dependents" tab, logout) because the URL change confirms the SPA has routed correctly. I used waitFor({ state: "visible" }) on the toast message after saving a dependent, because the toast is dynamically injected. I also used it for the logout link which only appears after clicking the dropdown.
  • Q13Difference between waitForSelector, waitForURL, and waitForTimeout?▾
    waitForSelector — Waits until an element appears/disappears in the DOM. Deprecated in favor of locator.waitFor().

    waitForURL — Waits until the page URL matches a string, regex, or predicate. Ideal for SPA navigation.

    waitForTimeout — A hard sleep. Last resort because it slows tests and doesn't wait for a condition. I used it sparingly (1s) after table updates where no better signal was available.
  • Q14How would you handle a flaky test caused by a slow network?▾
    (1) Increase timeout and actionTimeout in the config. (2) Add waitForURL or waitForResponse to wait for specific API calls to complete. (3) Use retries: 1 as a safety net. (4) Wait for DOM signals (spinner disappearing, row appearing) rather than hard timeouts. (5) Use page.waitForLoadState('networkidle') if the page makes many API calls during load.
  • Q15Why is waitForTimeout discouraged, and when might it be acceptable?▾
    It always waits the full duration even if the condition is met earlier (wasting time), and if the condition takes longer, the test still fails (not reliable). It's acceptable as a last resort when there's no DOM event, network request, or URL change to wait on — e.g. waiting for a CSS transition, or a third-party widget with no hookable events.

D — Assertions & Verification

  • Q16How did you verify that the dependent was successfully added?▾
    Two checks: (1) Name check — isDependentPresent("John Doe") searches the table for a cell containing the name and returns true/false. (2) Count check — recorded initialCount before adding and compared with newCount after, asserting expect(newCount).toBeGreaterThan(initialCount). This dual approach confirms both row content and row count increased.
  • Q17What assertion methods does Playwright provide? Which did you use?▾
    Two kinds: Web-first (auto-retrying): toBeVisible(), toHaveURL(), toHaveText(), toBeEnabled(), toHaveCount(). Generic (non-retrying): toBeTruthy(), toBeGreaterThan(), toEqual().

    I used: toBeVisible() for element checks, toHaveURL() for navigation, toBeTruthy() for isDependentPresent's boolean, and toBeGreaterThan() for the count comparison.
  • Q18Difference between expect(locator).toBeVisible() and locator.count() > 0?▾
    toBeVisible() is a web-first assertion — it auto-retries/polls the DOM until the timeout. If the element appears within that window, it passes.

    locator.count() is a one-shot evaluation — checks the DOM at that instant. If not rendered yet, returns 0 and the test fails immediately. Always prefer toBeVisible() or toHaveCount() for reliability.
  • Q19If the toast disappears before your assertion, how would you handle it?▾
    (1) Use locator.waitFor({ state: "visible" }) immediately after the triggering action — this is what I do in addDependent(). (2) Use page.waitForResponse() to wait for the API call and assert on the HTTP status (201/200) instead. (3) Listen via page.waitForSelector('.oxd-toast', { state: 'attached' }) which catches it even if it auto-dismisses quickly.
  • Q20How would you add a negative test — missing name shows validation error?▾
    Create a separate test that: (1) Navigates to the Dependents tab. (2) Clicks "Add". (3) Leaves name empty. (4) Selects a relationship. (5) Clicks "Save". (6) Asserts a validation message: await expect(page.locator('.oxd-input-field-error-message')).toHaveText('Required'). (7) Asserts the count did not increase.

E — Test Structure & Reporting

  • Q21Explain test.describe, test.beforeEach, and test.step.▾
    test.describe — Groups related tests under a common label. Useful for shared hooks and organisation.

    test.beforeEach — Runs before every test in the describe block. I use it to instantiate fresh page objects.

    test.step — Sub-divides a test into named steps. Each appears as a collapsible section in the HTML report and trace viewer, making it easy to see which step failed.
  • Q22If one test.step fails, do the remaining steps still run?▾
    No. When a step fails, the entire test exits and is marked as failed. Remaining steps are not executed. If you need independent checks, put them in separate tests. Steps are for sequential flows where later steps depend on earlier ones.
  • Q23How do you generate and view the Playwright HTML report?▾
    The config includes reporter: [["html", { open: "never" }], ["list"]]. After npx playwright test, Playwright generates the report in playwright-report/. View it with: npx playwright show-report. It opens a local web server showing pass/fail status, duration, screenshots, traces, and step breakdowns.
  • Q24Purpose of trace, screenshot, and video in config?▾
    trace: "on-first-retry" — Records DOM snapshots, network, console, and action timeline on the first retry. Open in trace viewer for step-by-step debugging.

    screenshot: "only-on-failure" — Captures a screenshot on failure, attached to the report.

    video: "retain-on-failure" — Records video for all tests but only keeps it on failure, avoiding bloated reports for passing tests.
  • Q25How would you run this test only in headed mode for a live demo?▾
    Run: npx playwright test --headed or npm run test:headed. Slow it down with npx playwright test --headed --slowMo=500 which adds a 500ms pause between each action so viewers can follow the flow.

F — Configuration

  • Q26Walk through your playwright.config.js — what does each key setting do?▾
    testDir — Folder containing test files.
    fullyParallel — Whether tests inside a file run in parallel.
    forbidOnly — Prevents accidental .only in CI.
    retries — Retries on failure (1 in CI, 0 locally).
    workers — Parallel worker processes.
    timeout — Max time per test (60s).
    baseURL — Prepended to relative goto() paths.
    actionTimeout — Max wait for auto-wait actions (15s).
    navigationTimeout — Max wait for navigation (30s).
    projects — Browser configurations to test against.
  • Q27Why set fullyParallel: false and workers: 1?▾
    The OrangeHRM demo is a shared environment. Running tests in parallel could cause conflicts (e.g. two tests adding/deleting dependents simultaneously). workers: 1 ensures only one browser runs at a time. For an isolated test environment, you'd enable both for faster execution.
  • Q28What is baseURL and how does it simplify navigation code?▾
    baseURL is a URL prefix applied to all relative paths in page.goto(). Instead of the full URL every time, I write page.goto("/web/index.php/auth/login"). It also makes it trivial to switch environments (dev, staging, prod) by changing one config value or overriding via an environment variable.
  • Q29How would you run the same test across Chrome, Firefox, and Safari?▾
    Add more entries to the projects array:
    projects: [ { name: "chromium", use: { ...devices["Desktop Chrome"] } }, { name: "firefox", use: { ...devices["Desktop Firefox"] } }, { name: "webkit", use: { ...devices["Desktop Safari"] } }, ]
    Then install all browsers: npx playwright install. Target one with --project=firefox.
  • Q30Where are credentials stored, and how would you secure them in production?▾
    Currently in utils/testData.js — fine for a demo site. In production: (1) Use environment variables: process.env.OHR_USERNAME. (2) Store in a .env file (gitignored) loaded with dotenv. (3) In CI, use secrets managers (GitHub Secrets, Azure Key Vault). (4) Never commit real credentials to version control.

G — Advanced / Scenario-Based

  • Q31How would you parameterise this test to add multiple dependents?▾
    Use Playwright's parameterised tests:
    const dependents = [ { name: "John Doe", relationship: "Child" }, { name: "Jane Smith", relationship: "Spouse" }, ]; for (const dep of dependents) { test(`add dependent: ${dep.name}`, async ({ page }) => { // login, navigate, add dep.name, verify, logout }); }
    Each iteration becomes a separate, independently-reported test.
  • Q32Cleanup strategy — afterEach, inside the test, or via API?▾
    Best: API cleanup — Use Playwright's request context in afterEach to call the REST API and delete the dependent. Fast, reliable, UI-independent.

    Acceptable: UI cleanup in afterEach — Call deleteFirstDependent(). Risk: if the test failed mid-way, page state might be unexpected.

    Least ideal: Inside the test — A failure skips the cleanup. Always prefer afterEach hooks because they run regardless of outcome.
  • Q33The demo site is shared — how does that affect your test?▾
    (1) Data pollution — Other users may modify data concurrently. Mitigation: assert on name presence not exact row count. (2) Slow responses — Use generous timeouts. (3) Data resets — Demo may reset periodically. (4) No parallel execution — Run sequentially. For production, use an isolated environment.
  • Q34How would you integrate this into CI/CD (e.g. GitHub Actions)?▾
    Create .github/workflows/playwright.yml:
    name: Playwright Tests on: [push, pull_request] jobs: test: runs-on: ubuntu-latest steps: - uses: actions/checkout@v4 - uses: actions/setup-node@v4 with: { node-version: 20 } - run: npm ci - run: npx playwright install --with-deps chromium - run: npx playwright test - uses: actions/upload-artifact@v4 if: failure() with: name: playwright-report path: playwright-report/
  • Q35What is a fixture? How would you create one for the logged-in state?▾
    A fixture is a reusable setup/teardown mechanism injected into tests:
    const test = base.extend({ loggedInPage: async ({ page }, use) => { const lp = new LoginPage(page); await lp.navigate(); await lp.login("Admin", "admin123"); await use(page); // test runs here // teardown after }, }); test("my test", async ({ loggedInPage }) => { // Already logged in! });
    Avoids repeating login code and ensures consistent setup/teardown.
  • Q36Difference between page.goto() and page.navigate()?▾
    In Playwright, there is no page.navigate(). The correct method is page.goto(url). This is a common confusion from Selenium (which uses driver.navigate().to(url)). In our project, navigate() is a custom method name on LoginPage that internally calls page.goto().
  • Q37How would you handle file upload / download?▾
    Upload: locator.setInputFiles('path/to/file.pdf') or page.on('filechooser') for non-standard triggers.

    Download:
    const downloadPromise = page.waitForEvent('download'); await page.locator('#export-btn').click(); const download = await downloadPromise; await download.saveAs('./downloads/' + download.suggestedFilename());
  • Q38Playwright's retry mechanisms vs re-running the entire test?▾
    Web-first assertion retries: Methods like toBeVisible() auto-poll the DOM until timeout — micro-level retry within a single action.

    Test-level retries: The retries config re-runs the entire test from scratch (fresh browser context). Handles broader flakiness from network/timing issues.

    They work at different levels: assertion retries handle momentary DOM delays; test retries handle environmental flakiness.
  • Q39Testing on a mobile viewport — what changes?▾
    Add a mobile project: { name: "mobile", use: { ...devices["iPhone 14"] } }. This sets viewport, user agent, and touch events. You may need to: (1) Handle hamburger menus — sidebar may collapse. (2) Adjust locators for mobile layout. (3) Account for touch-based dropdowns.
  • Q40How would you use Playwright's API request context for test data setup?▾
    Playwright provides request as a built-in fixture:
    test.beforeEach(async ({ request }) => { await request.post('/api/v2/auth/login', { data: { username: 'Admin', password: 'admin123' } }); await request.post('/api/v2/pim/employees/7/dependents', { data: { name: 'John Doe', relationshipType: 'child' } }); });
    Much faster than UI setup and less flaky. Also great for cleanup in afterEach.

H — Debugging & Troubleshooting

  • Q41Passes locally, fails in CI — first three things you check?▾
    (1) Timeouts — CI runners are slower. Increase all timeouts. (2) Browser dependencies — Use npx playwright install --with-deps for OS-level libraries. (3) Environment differences — Check baseURL, credentials, network access. Open the trace artifact from the failed CI run for a visual replay.
  • Q42How do you use the trace viewer to diagnose a failure?▾
    Open the trace file: npx playwright show-trace trace.zip. It shows: (1) Action timeline — every click, fill, navigation. (2) DOM snapshots — before/after each action. (3) Network log — all HTTP requests. (4) Console log — JS errors. Click any action to see the exact DOM state when it failed.
  • Q43What is --debug mode and how does it help?▾
    npx playwright test --debug opens the Playwright Inspector: (1) Pauses at the first action. (2) Step-over button for one action at a time. (3) Highlights target elements. (4) Locator picker — hover over elements for suggested selectors. (5) Call log showing internal Playwright operations. Invaluable during development.
  • Q44How would you capture console errors or network failures?▾
    Console errors:
    page.on('console', msg => { if (msg.type() === 'error') console.log('Browser error:', msg.text()); });
    Network failures:
    page.on('requestfailed', req => { console.log('Failed:', req.url(), req.failure().errorText); });
    Place in beforeEach. Collect in an array and assert no errors in afterEach.
  • Q45A locator broke after a UI redesign — describe your fix process.▾
    (1) Debug mode: npx playwright test --debug to see the page live. (2) Locator picker: hover over the element for a suggested selector. (3) Inspect DOM: check what changed (tag, class, parent). (4) Update page object: fix the locator in the single file. (5) Run test: verify it passes. (6) Audit: check other locators in the same page object. (7) Commit: reference the UI change. Because of POM, only one file needs updating.
