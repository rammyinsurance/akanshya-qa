/**
 * Page Object Model for the OrangeHRM Login Page.
 *
 * Encapsulates all locators and actions related to the login screen.
 */
class LoginPage {
  /**
   * @param {import('@playwright/test').Page} page
   */
  constructor(page) {
    this.page = page;

    // --- Locators ---
    this.usernameInput = page.locator('input[name="username"]');
    this.passwordInput = page.locator('input[name="password"]');
    this.loginButton   = page.locator('button[type="submit"]');
    this.errorMessage  = page.locator(".oxd-alert-content--error");
    this.brandingLogo  = page.locator(".orangehrm-login-branding img");
  }

  // --- Actions ---

  /** Navigate to the login page. */
  async navigate() {
    await this.page.goto("/web/index.php/auth/login");
    await this.brandingLogo.waitFor({ state: "visible" });
  }

  /**
   * Perform login with provided credentials.
   * @param {string} username
   * @param {string} password
   */
  async login(username, password) {
    await this.usernameInput.fill(username);
    await this.passwordInput.fill(password);
    await this.loginButton.click();
  }

  /** Returns the error message text if login fails. */
  async getErrorMessage() {
    await this.errorMessage.waitFor({ state: "visible" });
    return this.errorMessage.textContent();
  }
}

module.exports = { LoginPage };

/**
 * Page Object Model for the OrangeHRM Dashboard Page.
 *
 * Handles top-level navigation (sidebar menu items) and logout.
 */
class DashboardPage {
  /**
   * @param {import('@playwright/test').Page} page
   */
  constructor(page) {
    this.page = page;

    // --- Locators ---
    this.sidebarMyInfo    = page.locator('a.oxd-main-menu-item',
                              { hasText: "My Info" });
    this.userDropdown     = page.locator(".oxd-userdropdown-tab");
    this.logoutLink       = page.locator(
      'a[href="/web/index.php/auth/logout"]');
    this.dashboardHeading = page.locator("h6",
                              { hasText: "Dashboard" });
  }

  /** Wait until the dashboard is fully loaded. */
  async waitForDashboard() {
    await this.dashboardHeading.waitFor({
      state: "visible", timeout: 15000
    });
  }

  /** Click "My Info" in the left sidebar. */
  async navigateToMyInfo() {
    await this.sidebarMyInfo.click();
    await this.page.waitForURL("**/viewPersonalDetails/**",
      { timeout: 15000 });
  }

  /** Perform logout via the user dropdown. */
  async logout() {
    await this.userDropdown.click();
    await this.logoutLink.waitFor({ state: "visible" });
    await this.logoutLink.click();
    await this.page.waitForURL("**/auth/login",
      { timeout: 15000 });
  }
}

module.exports = { DashboardPage };

/**
 * Page Object Model for the OrangeHRM Dependents tab
 * under My Info → Dependents.
 */
class DependentsPage {
  constructor(page) {
    this.page = page;

    // --- Tab Locator ---
    this.dependentsTab = page.locator(
      'a[href*="viewDependents"]', { hasText: "Dependents" });

    // --- Form Locators ---
    this.addButton    = page.locator("button", { hasText: "Add" });
    this.saveButton   = page.locator(
      'button[type="submit"]', { hasText: "Save" });
    this.cancelButton = page.locator("button", { hasText: "Cancel" });

    // Form fields inside .oxd-form
    this.nameInput = page.locator(
      '.oxd-form .oxd-input-group:has(label:text("Name")) input');
    this.relationshipDropdown = page.locator(
      '.oxd-form .oxd-input-group:has(label:text("Relationship")) .oxd-select-text');
    this.dateOfBirthInput = page.locator(
      '.oxd-form .oxd-input-group:has(label:text("Date of Birth")) input');

    // --- Table Locators ---
    this.dependentsTable = page.locator(".oxd-table");
    this.tableRows       = page.locator(
      ".oxd-table .oxd-table-body .oxd-table-row");
    this.toastMessage    = page.locator(".oxd-toast");

    // --- Delete Locators ---
    this.deleteCheckbox      = page.locator(
      ".oxd-table-body .oxd-table-row .oxd-checkbox-input");
    this.deleteSelectedButton = page.locator(
      "button", { hasText: "Delete Selected" });
    this.confirmDeleteButton  = page.locator(
      ".orangehrm-modal-footer button",
      { hasText: "Yes, Delete" });
  }

  /** Navigate to the Dependents tab within My Info. */
  async navigateToDependentsTab() {
    await this.dependentsTab.click();
    await this.page.waitForURL("**/viewDependents/**",
      { timeout: 15000 });
    await this.addButton.waitFor({
      state: "visible", timeout: 10000
    });
  }

  /** Get the count of dependents currently in the table. */
  async getDependentCount() {
    await this.page.waitForTimeout(1000);
    return await this.tableRows.count();
  }

  /**
   * Add a new dependent.
   * @param {string} name
   * @param {string} relationship  - "Child", "Spouse", or "Other"
   * @param {string} [dateOfBirth] - YYYY-MM-DD (optional)
   */
  async addDependent(name, relationship, dateOfBirth) {
    await this.addButton.click();

    // Fill the name
    await this.nameInput.waitFor({ state: "visible", timeout: 5000 });
    await this.nameInput.fill(name);

    // Select relationship from dropdown
    await this.relationshipDropdown.click();
    const option = this.page.locator(
      `.oxd-select-dropdown .oxd-select-option`,
      { hasText: relationship }
    );
    await option.click();

    // Fill date of birth if provided
    if (dateOfBirth) {
      await this.dateOfBirthInput.clear();
      await this.dateOfBirthInput.fill(dateOfBirth);
      await this.page.keyboard.press("Escape");
    }

    // Save & wait for success toast
    await this.saveButton.click();
    await this.toastMessage.waitFor({
      state: "visible", timeout: 10000
    });
  }

  /** Check whether a dependent with the given name exists. */
  async isDependentPresent(name) {
    await this.page.waitForTimeout(1000);
    const row = this.page.locator(
      `.oxd-table-body .oxd-table-row .oxd-table-cell:has-text("${name}")`
    );
    return (await row.count()) > 0;
  }

  /** Delete the first dependent using checkbox + Delete Selected. */
  async deleteFirstDependent() {
    const firstCheckbox = this.deleteCheckbox.first();
    await firstCheckbox.click();
    await this.deleteSelectedButton.click();
    await this.confirmDeleteButton.waitFor({ state: "visible" });
    await this.confirmDeleteButton.click();
    await this.toastMessage.waitFor({
      state: "visible", timeout: 10000
    });
  }
}

module.exports = { DependentsPage };

// @ts-check
const { test, expect } = require("@playwright/test");
const { LoginPage }      = require("../pages/LoginPage");
const { DashboardPage }  = require("../pages/DashboardPage");
const { DependentsPage } = require("../pages/DependentsPage");
const { TestData }       = require("../utils/testData");

/**
 * Orange HRM – Add Dependent & Verify (End-to-End)
 *
 * Flow:
 *   1. Open login page
 *   2. Login with valid credentials
 *   3. Navigate to My Info
 *   4. Click the Dependents tab
 *   5. Add a new dependent
 *   6. Verify the dependent was added
 *   7. Logout
 */
test.describe("OrangeHRM – Dependents Management", () => {
  /** @type {LoginPage} */
  let loginPage;
  /** @type {DashboardPage} */
  let dashboardPage;
  /** @type {DependentsPage} */
  let dependentsPage;

  test.beforeEach(async ({ page }) => {
    loginPage      = new LoginPage(page);
    dashboardPage  = new DashboardPage(page);
    dependentsPage = new DependentsPage(page);
  });

  test("should add a dependent and verify, then logout",
    async ({ page }) => {

    // ── Step 1: Open Login Page ──────────────────────
    await test.step("Open the login page", async () => {
      await loginPage.navigate();
      await expect(loginPage.usernameInput).toBeVisible();
    });

    // ── Step 2: Login ────────────────────────────────
    await test.step("Login with valid credentials", async () => {
      await loginPage.login(
        TestData.credentials.username,
        TestData.credentials.password
      );
      await dashboardPage.waitForDashboard();
      await expect(dashboardPage.dashboardHeading).toBeVisible();
    });

    // ── Step 3: Navigate to My Info ──────────────────
    await test.step("Click My Info from the sidebar", async () => {
      await dashboardPage.navigateToMyInfo();
      await expect(page).toHaveURL(/viewPersonalDetails/);
    });

    // ── Step 4: Click Dependents Tab ─────────────────
    await test.step("Click the Dependents tab", async () => {
      await dependentsPage.navigateToDependentsTab();
      await expect(page).toHaveURL(/viewDependents/);
    });

    // ── Step 5: Record count & Add dependent ─────────
    const initialCount =
      await dependentsPage.getDependentCount();

    await test.step("Add a new dependent", async () => {
      await dependentsPage.addDependent(
        TestData.dependent.name,
        TestData.dependent.relationship,
        TestData.dependent.dateOfBirth
      );
    });

    // ── Step 6: Verify dependent was added ───────────
    await test.step("Verify dependent in the table", async () => {
      const isPresent =
        await dependentsPage.isDependentPresent(
          TestData.dependent.name
        );
      expect(isPresent).toBeTruthy();

      const newCount =
        await dependentsPage.getDependentCount();
      expect(newCount).toBeGreaterThan(initialCount);
    });

    // ── Step 7: Logout ───────────────────────────────
    await test.step("Perform logout", async () => {
      await dashboardPage.logout();
      await expect(loginPage.usernameInput).toBeVisible();
    });
  });
});
